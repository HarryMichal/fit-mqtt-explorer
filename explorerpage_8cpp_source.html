44 {
45  auto curr_row = this->ui->messageHistoryList->row(current);
46  auto prev_row = this->ui->messageHistoryList->row(previous);
47 
48  // No need to refresh the view when no other item is selected.
49  if (curr_row == prev_row) {
50  return;

52 
53  // Hack alert: In order to prevent losing the preview of a message that

153 void ExplorerPage::addMessages(QList<Message*> msgs)
154 {
155  for (auto it = msgs.begin(); it != msgs.end(); it++) {
156  std::string msg = (*it)->getMessage();
157  QByteArray data(msg.c_str(), msg.length());
158  QListWidgetItem *new_item = new QListWidgetItem;
159  QPixmap pixmap;
160 
161  if (pixmap.loadFromData(data, "JPG") || pixmap.loadFromData(data, "PNG")) {
162  // The payload is a pixmap (picture)
163  msg = "<binary-message::picture>";
164  } else {
165  // Truncate messages longer than a certain number of characters
166  // If truncated, add three dots to signalize it.
167 
168  if (msg.length() > 26) {
169  msg.resize(26);
170  msg += "...";
171  }
172  }
173 
174  new_item->setText(QString::fromStdString(msg));
175  // Based on message type set background color. I took some colors that
176  // looked nice in Krita.
177  if ((*it)->msg_type == MessageType::RECEIVED) {
178  new_item->setBackground(QBrush(QColor(184, 244, 245)));
179  } else {
180  new_item->setBackground(QBrush(QColor(255, 125, 125)));
181  }
182 
183  this->ui->messageHistoryList->insertItem(0, new_item);
184 
185  // Respect the set message capacity
186  for (int i = this->ui->messageHistoryList->count(); i > this->message_capacity; i--) {
187  delete this->ui->messageHistoryList->takeItem(i-1);
188  }
189  }
190 }

192 void ExplorerPage::addTopic(const QString topic, const bool root)
193 {
194  QStandardItem *parent = this->topics_tree_model.invisibleRootItem();
195 
196  // The root node is always called the same as the server. The rest of the
197  // nodes are always a child of this node.
198  if (!root) {
199  parent = parent->child(0);
200  }
201 
202  // To build up the tree, we need to check every parent topic in the chain.
203  auto topics = topic.split("/");
204 
205  for (QList<QString>::iterator i = topics.begin(); i != topics.end(); i++) {
206  bool found = false;
207 
208  // If we want to add an item, we need to find its parent first. The
209  // item is then added to it.
210  for (int row = 0; row < parent->rowCount(); row++) {
211  if (*i != parent->child(row)->text()) {
212  continue;
213  }
214 
215  // Found topic with the same name -> new parent
216  parent = parent->child(row);
217  found = true;
218  break;
219  }
220 
221  if (found) {
222  continue;
223  }
224 
225  QStandardItem *new_topic_node = new QStandardItem();
226  new_topic_node->setText(*i);
227  parent->appendRow(new_topic_node);
228  parent = new_topic_node;
229  this->topics_tree_model.sort(0);
230  }
231 }
