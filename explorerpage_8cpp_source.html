151 void ExplorerPage::addMessages(QList<QString> msgs)
152 {
153  for (auto it = msgs.begin(); it != msgs.end(); it++) {
154  QByteArray data(it->toStdString().c_str(), it->length());
155  QPixmap pixmap;
156  QString msg;
157 
158  if (pixmap.loadFromData(data, "JPG") || pixmap.loadFromData(data, "PNG")) {
159  // The payload is a pixmap (picture)
160  msg = "<binary-message::picture>";
161  } else {
162  // Truncate messages longer than a certain number of characters
163  // If truncated, add three dots to signalize it.
164  msg = *it;
165 
166  if (msg.length() > 26) {
167  msg.truncate(26);
168  msg += "...";
169  }
170  }
171 
172  this->ui->messageHistoryList->insertItem(0, msg);
173  // Respect the set message capacity
174  for (int i = this->ui->messageHistoryList->count(); i > this->message_capacity; i--) {
175  delete this->ui->messageHistoryList->takeItem(i-1);
176  }
177  }
178 }

180 void ExplorerPage::addTopic(const QString topic, const bool root)
181 {
182  QStandardItem *parent = this->topics_tree_model.invisibleRootItem();
183 
184  // The root node is always called the same as the server. The rest of the
185  // nodes are always a child of this node.
186  if (!root) {
187  parent = parent->child(0);
188  }
189 
190  // To build up the tree, we need to check every parent topic in the chain.
191  auto topics = topic.split("/");
192 
193  for (QList<QString>::iterator i = topics.begin(); i != topics.end(); i++) {
194  bool found = false;
195 
196  // If we want to add an item, we need to find its parent first. The
197  // item is then added to it.
198  for (int row = 0; row < parent->rowCount(); row++) {
199  if (*i != parent->child(row)->text()) {
200  continue;
201  }
202 
203  // Found topic with the same name -> new parent
204  parent = parent->child(row);
205  found = true;
206  break;
207  }
208 
209  if (found) {
210  continue;
211  }
212 
213  QStandardItem *new_topic_node = new QStandardItem();
214  new_topic_node->setText(*i);
215  parent->appendRow(new_topic_node);
216  parent = new_topic_node;
217  this->topics_tree_model.sort(0);
218  }
219 }
