99 void MessageStore::addMessage(const mqtt::const_message_ptr msg, MessageType type)
100 {
101  // This works for both scenarios: topic is known and topic is unknown.
102  // Using the []operator to get an item causes the creation of a default
103  // item than can be used to add the new message payload.
104  auto qtopic = QString::fromStdString(msg->get_topic());
105  QList<Message*> msg_list = this->new_messages[qtopic];
106 
107  Message* new_msg = new Message(msg, type);
108 
109  // Check if a received message matches a sent message by comparing the
110  // payload and arrival date. A matching received message can be discarded.
111  if (new_msg->msg_type == MessageType::RECEIVED) {
112  QMutableListIterator<Message*> it(msg_list);
113  it.toBack();
114  while (it.hasPrevious()) {
115  Message* msg = it.previous();
116  std::chrono::duration<double> time_diff;
117 
118  time_diff = new_msg->arrival_date - msg->arrival_date;
119 
120  // The time difference between messages is too big. We can skip the
121  // rest of the messages and add the new message.
122  if (time_diff.count() > 1) {
123  break;
124  }
125 
126  // An already matched sent message was found. No unmatched messages
127  // should be any further. We can skip the rest of the messages and
128  // add the new message.
129  if (msg->msg_type == MessageType::SENT_MATCHED) {
130  break;
131  }
132 
133  // No need to compare two received messages.
134  if (msg->msg_type == MessageType::RECEIVED) {
135  continue;
136  }
137 
138  // Messages don't match -> definitely a different message
139  if (msg->getMessage() != new_msg->getMessage()) {
140  continue;
141  }
142 
143  // We found a matching sent message within a specified time frame.
144  // The received message is most likely an echo from the broker.
145  msg->msg_type = MessageType::SENT_MATCHED;
146  delete new_msg;
147  return;
148  }
149  }
150 
151  msg_list.append(new_msg);
152 
153  this->new_messages.insert(qtopic, msg_list);
154 }
