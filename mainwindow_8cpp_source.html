211 void MainWindow::sendFile(mqtt::string topic, QString file_name)
212 {
213  // Read the file and use its content as a payload

215 
216  if (!file.exists()) {
217  return;

227  mqtt::const_message_ptr msg = mqtt::make_message(topic, payload, 0, true);
